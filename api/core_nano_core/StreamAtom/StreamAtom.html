StreamAtom(
  Stream<T> stream, {
  AsyncState<T> initial = const AsyncLoading(),
  String? label,
}) : super(initial, label: label) {
  _subscription = stream.listen(
    (data) => Nano.action(() => set(AsyncData<T>(data))),
    onError: (e, s) {
      Nano.action(() {
        set(AsyncError<T>(e, s));
        Nano.observer.onError(this, e, s);
      });
    },
  );
}
