get<T> method

T get<T>()

Retrieves the registered instance of type T.

Throws NanoException if the type is not registered.

T get<T>() {
  final initContext = Nano.initContext;
  if (initContext != null && !initContext.isValid) {
    debugPrint(
      '‚ö†Ô∏è [Nano] DI Warning: You are resolving ${T.toString()} after an await in onInit.\n'
      'This is dangerous as the Scope/Registry might have changed between awaits.\n'
      'üëâ Fix: Resolve all dependencies synchronously at the start of onInit.',
    );
  }

  if (_lookupStack.contains(T)) {
    throw NanoException(
      "Circular dependency detected while resolving '${T.toString()}'.\n"
      "üëâ Chain: ${_lookupStack.join(' -> ')} -> ${T.toString()}",
    );
  }

  _lookupStack.add(T);
  try {
    // 1. Check existing instances
    if (_services.containsKey(T)) {
      return _services[T] as T;
    }

    // 2. Check Factories (create new)
    if (_factories.containsKey(T)) {
      final instance = _factories[T]!(this) as T;
      if (instance is NanoLogicBase) {
        instance.initializeDynamic(null);
      }
      return instance;
    }

    // 3. Check Lazy Singletons (create, cache, return)
    if (_lazySingletons.containsKey(T)) {
      final instance = _lazySingletons[T]!(this);
      _services[T] = instance; // Cache it!
      _lazySingletons.remove(T); // Optimization: remove from lazy map

      if (instance is NanoLogicBase) {
        instance.initializeDynamic(null);
      }

      return instance as T;
    }

    // 4. Try Parent
    if (parent != null) {
      return parent!.get<T>();
    }

    throw NanoException(
      "Service of type '${T.toString()}' not found in the current Scope.\n"
      "üëâ Fix: Ensure you added '${T.toString()}' to the 'modules' list in your Scope widget.",
    );
  } finally {
    _lookupStack.removeLast();
  }
}
